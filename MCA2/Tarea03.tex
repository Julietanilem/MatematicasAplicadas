\begin{itemize}[format=\textbf]

\item Evalúe $f(1,1,1)$.

\begin{align*}
f(1,1,1)
&= \sqrt{1} +\sqrt{1} +\sqrt{1} + \ln{(4-1^2-1^2-1^2)} \\
&= 1+1+1+ \ln{(4-1-1-1)} \\
&= 1+1+1+ \ln{1} \\
&= 1+1+1+0 \\
&= 3
\end{align*}

\item Determine y describa el dominio de $f(x, y, z)$.

\end{itemize}
